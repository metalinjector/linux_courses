\begin{frame}[fragile]
	\frametitle{Перенаправление ввода/вывода}

	\begin{itemize}

		\item ``<<'' -- Here-документ

		\item ``<>'' -- Открывает файловый дескриптор из файла/другого дескритора
			\begin{block}{Пример}
				\begin{lstlisting}
exec 3<>test.txt; echo test >&3;  cat <test.txt
\end{lstlisting}
			\end{block}
			
		\item ``n<\&-'' -- Закрывает файловый дескриптор
			\begin{block}{Пример}
				\begin{lstlisting}
exec 3<&-; echo test >&3
\end{lstlisting}
			\end{block}
			
		\item ``|'' -- pipe
	\end{itemize}

\end{frame}
