\begin{frame}[fragile]
\frametitle{Примеры использования while}

\begin{block}{Перебираем аргументы.}
\begin{lstlisting}[language=bash,frame=single]
while [[ -n $1 ]]
do
    echo $1
    shift # rename the positional parameters $N+1,$N+2 ... to $1,$2 (N=1)
done
\end{lstlisting}
\end{block}
