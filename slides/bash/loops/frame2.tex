\begin{frame}[fragile]
  \frametitle{Циклы for}
  \begin{enumerate}
    \item Стандартная форма
\begin{lstlisting}[language=sh,frame=single]
  for x in list 
  do
    op1
    op2
  done
\end{lstlisting}
    \item Арифметическая форма
\begin{lstlisting}[language=sh,frame=single]
  for (( expr1 ; expr2 ; expr3 )) 
  do 
    op1
    op2
  done
\end{lstlisting}
  \end{enumerate}
\end{frame}
