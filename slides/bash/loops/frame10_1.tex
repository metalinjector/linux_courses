\begin{frame}[fragile]
\frametitle{ Цикл until. Пример.}
  \begin{block}{Ожидаем хост после перезагрузки.}
    \begin{lstlisting}[language=sh,frame=single]
until ping -q -c 3  $host 1>/dev/null 2>&1 && nc -z $host 22
do 
   sleep 1
   echo unavailable;
done
    \end{lstlisting}
\end{block}
\end{frame}
