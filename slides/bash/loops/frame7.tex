\begin{frame}[fragile]
\frametitle{ Циклы for. Примеры.}
  \begin{block}{Из аргументов.}
    \begin{lstlisting}[language=sh,frame=single]
for arg
 do echo $arg 
done
    \end{lstlisting}
  \end{block}
  \begin{block}{Из переменной. Запись одной строкой.}
    \begin{lstlisting}[language=sh,frame=single]
for name in $users ; do echo $name ; done
    \end{lstlisting}
  \end{block}
\end{frame}
