\begin{frame}[fragile]
	\frametitle{Функции: синтаксис}
	\begin{itemize}
		\item Классический синаксис: 
			\begin{lstlisting}
function function_name {
  command...
}
\end{lstlisting}
		\item Портабельный (C-style):
			\begin{lstlisting}
function_name()
{
  command...
} 
\end{lstlisting}

		\item Однострочный:
			\begin{lstlisting}
function_name () { command... ;}
\end{lstlisting}
  \end{itemize}
\end{frame}
