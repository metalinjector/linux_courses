\begin{frame}[fragile]
	\frametitle{Функции: синтаксис}
	\begin{itemize}
		\item Классический синаксис c ключевым словом \alert{function}: 
                        \begin{lstlisting}[language=bash,frame=single]
function function_name {
  command...
}
\end{lstlisting}
		\item Портабельный (C-style):
			\begin{lstlisting}
function_name()
{
  command...
} 
\end{lstlisting}

		\item Однострочный:
                        \begin{lstlisting}[language=bash,frame=single]
function_name () { command... ;}
\end{lstlisting}
  \end{itemize}
\end{frame}
